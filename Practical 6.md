import cv2 as cv
import numpy as np

cap = cv.VideoCapture(0)

while True:
    ret, frame = cap.read()
    hsv = cv.cvtColor(frame, cv.COLOR_BGR2HSV)

    lower = np.array([83,50,50])
    upper = np.array([103, 255, 255])
    mask = cv.inRange(hsv, lower, upper)
    
    contours, _ = cv.findContours(mask, cv.RETR_TREE, cv.CHAIN_APPROX_SIMPLE)
    print(len(contours))
    counter = 0
    font = cv.FONT_HERSHEY_SIMPLEX
    for contour in contours:
        if cv.contourArea(contour) > 1000: # Filter out small areas, pixel2
            counter = counter + 1
            (x,y), radius = cv.minEnclosingCircle(contour)
            center = (int(x), int(y))
            radius = int(radius)
            cv.circle(frame, center, radius, (0, 255, 0), 2)
            cv.circle(mask, center, radius, (0, 255, 0), 2)
    
    cv.putText(frame, 'Object Detected: ' + str(counter),(50,50), font, 2, (0,0,255), 2, cv.LINE_AA)
    cv.imshow("The Video", frame)
    cv.imshow("Mask", mask)
    
    if cv.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv.destroyAllWindows()
